function PersonajesModule.obtenerPersonajeAleatorio()
	local nombres = {}
	for nombre, _ in pairs(PersonajesModule.personajes) do
		table.insert(nombres, nombre)
	end
	return nombres[math.random(1, #nombres)]
end

function PersonajesModule.obtenerEquipoBalanceado()
	local equipos = {
		Equipo1 = 0,
		Equipo2 = 0
	}

	for _, player in ipairs(game:GetService("Players"):GetPlayers()) do
		local equipo = player:GetAttribute("Equipo")
		if equipo and equipos[equipo] then
			equipos[equipo] = equipos[equipo] + 1
		end
	end

	if equipos.Equipo1 <= equipos.Equipo2 then
		return "Equipo1"
	else
		return "Equipo2"
	end
end

return PersonajesModule
